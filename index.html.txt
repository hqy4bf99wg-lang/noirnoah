<script>
/* accordion */
const sections=document.querySelectorAll('.section');
sections.forEach(sec=>{
  sec.querySelector('button').onclick=e=>{
    e.stopPropagation();
    sections.forEach(s=>s!==sec&&s.classList.remove('open'));
    sec.classList.toggle('open');
  };
});

/* theme toggle */
document.body.addEventListener('click',e=>{
  if(e.target.closest('a,button,.clock')) return;
  document.body.classList.toggle('inverted');
});

/* clock */
const clock=document.getElementById('clock');
function updateClock(){
  const n=new Date();
  clock.textContent=
  `${String(n.getHours()).padStart(2,'0')}:${String(n.getMinutes()).padStart(2,'0')}:${String(n.getSeconds()).padStart(2,'0')}`;
}
updateClock();
setInterval(updateClock,1000);

/* floating clock (desktop only) */
if (window.matchMedia("(pointer: fine)").matches) {
  document.addEventListener("mousemove", e=>{
    const x=(e.clientX/window.innerWidth-.5)*10;
    const y=(e.clientY/window.innerHeight-.5)*10;
    clock.style.transform=`translate(calc(-50% + ${x}px), ${y}px)`;
  });
}

/* eye logic */
const scale=8, blinkDuration=90, decay=.28;
const pathOpen="M10 50 C 30 20, 70 20, 90 50 C 70 80, 30 80, 10 50 Z";
const pathClosed="M10 50 C 30 50, 70 50, 90 50 C 70 50, 30 50, 10 50 Z";

class EyeGraphic extends HTMLElement{
  target={x:0,y:0};direction={x:0,y:0};
  connectedCallback(){
    this.attachShadow({mode:"open"});
    this.shadowRoot.append(
      document.getElementById("eye-graphic-template").content.cloneNode(true)
    );
    this.pupil=this.shadowRoot.querySelector('[data-id="pupil"]');
    this.lids=this.shadowRoot.querySelector('[data-id="lids"]');
    this.clip=this.shadowRoot.querySelector('[data-id="lids-clip"]');
    this.anim=this.shadowRoot.querySelector('[data-id="animation"]');
    this.lids.setAttribute("d",pathOpen);
    this.clip.setAttribute("d",pathOpen);
  }
  animate(f,t){
    this.lids.setAttribute("d",f);
    this.clip.setAttribute("d",f);
    this.anim.setAttribute("from",f);
    this.anim.setAttribute("to",t);
    this.anim.beginElement();
    setTimeout(()=>{this.lids.setAttribute("d",t);this.clip.setAttribute("d",t)},blinkDuration);
  }
  blinkOnce(){
    this.animate(pathOpen,pathClosed);
    setTimeout(()=>this.animate(pathClosed,pathOpen),blinkDuration);
  }
  look(x,y){this.target={x,y}}
  update(){
    this.direction.x+=(this.target.x*scale-this.direction.x)*decay;
    this.direction.y+=(this.target.y*scale-this.direction.y)*decay;
    this.pupil.style.transform=`translate3d(${this.direction.x}px,${this.direction.y}px,0)`;
  }
}
customElements.define("eye-graphic",EyeGraphic);

const eye=document.querySelector("eye-graphic");
(function loop(){eye.update();requestAnimationFrame(loop)})();
setInterval(()=>eye.look(Math.random()*1.4-.7,Math.random()*1.4-.7),4200);
setInterval(()=>eye.blinkOnce(),6800);
</script>
